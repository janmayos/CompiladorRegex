char introduce_operador()
{
   char op;
   bool error;
   do
   {
      cout << "Introduzca el operador: ";
      cin >> op;
      error = false;
      switch (op)
      {
         case '+':
         case '-':
         case '*':
         case '/':
         case '%':
            break;
         default:
            cout << "ERROR: El operador introducido no es valido.\n";
            error = true;
            break;
      }
   }while (error);

   return op;
}
